21 struct timeval diff_timeval(struct timeval start, struct timeval end)
22 {
23  static const long int factor = 1000000;
24  if (end.tv_usec < start.tv_usec)
25  {
26  int seconds = (start.tv_usec - end.tv_usec) / factor + 1;
27  start.tv_usec -= factor * seconds;
28  start.tv_sec += seconds;
29  }
30  if (end.tv_usec - start.tv_usec > factor)
31  {
32  int seconds = (end.tv_usec - start.tv_usec) / factor;
33  start.tv_usec += factor * seconds;
34  start.tv_sec -= seconds;
35  }
36  struct timeval result;
37  result.tv_sec = end.tv_sec - start.tv_sec;
38  result.tv_usec = end.tv_usec - start.tv_usec;
39 
40  return result;
41 }
