217 int icmp6_packet_checksum(icmp6_packet * packet)
218 {
219  int success = check_pointer(packet);
220 
221  if (success == 0)
222  {
223  icmp6_checksum_packet temporary_packet;
224  memset(&temporary_packet, 0, sizeof temporary_packet);
225  fake_ip6_hdr_init(&temporary_packet.ip_header, &packet->ip_header);
226  copy_icmp6(&temporary_packet.icmp_header, &packet->icmp_header);
227 
228  icmp6_hdr * icmp_header = &temporary_packet.icmp_header;
229  icmp_header->icmp6_cksum = 0;
230  temporary_packet.icmp_header.icmp6_cksum = checksum(&temporary_packet, (sizeof temporary_packet) / 2);
231  u_int16_t check = checksum(&temporary_packet, (sizeof temporary_packet) / 2);
232  if (check != 0)
233  {
234  errno = ECANCELED;
235  success = -2;
236  }
237  else
238  packet->icmp_header.icmp6_cksum = temporary_packet.icmp_header.icmp6_cksum;
239  }
240 
241  return success;
242 }
