149 int iphdr_checksum(iphdr * header)
150 {
151  int success = check_iphdr(header);
152 
153  if (success == 0)
154  {
155  u_int16_t old_sum = header->check;
156  header->check = 0;
157  header->check = checksum(header, (sizeof *header) / 2);
158  u_int16_t check = checksum(header, (sizeof *header) / 2);
159  if (check != 0)
160  {
161  header->check = old_sum;
162  errno = ECANCELED;
163  success = -3;
164  }
165  }
166 
167  return success;
168 }
