95 int icmp_checksum(icmphdr * header)
96 {
97  int success = check_pointer(header);
98 
99  if (success == 0)
100  {
101  u_int16_t old_sum = header->checksum;
102  header->checksum = 0;
103  header->checksum = checksum(header, (sizeof *header) / 2);
104  u_int16_t check = checksum(header, (sizeof *header) / 2);
105  if (check != 0)
106  {
107  header->checksum = old_sum;
108  errno = ECANCELED;
109  success = -2;
110  }
111  }
112 
113  return success;
