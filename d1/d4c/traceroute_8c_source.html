98 int traceroute_icmp_v4(const char * hostname, int hops_max, int attempts_number, int packet_size, const struct timeval * wait_time)
99 {
100  int sockfd;
101  struct sockaddr_in address;
102  char buffer[packet_size];
103  for (int i = 0; i < packet_size; i++)
104  buffer[i] = 0;
105  icmp4_packet * packet = (icmp4_packet *) buffer;
106  int success = traceroute_icmp_v4_init(hostname, &sockfd, &address, packet, packet_size);
107 
108  if (success != 0)
109  return success;
110 
111  print_traceroute_greeting(AF_INET, (struct sockaddr *) &address, hops_max, packet_size);
112 
113  int current_type = -1;
114  for (int ttl = 1; current_type != ICMP_ECHOREPLY && ttl <= hops_max; ttl++)
115  {
116  int printed = 1;
117  struct timeval times[attempts_number];
118  memset(times, 0, attempts_number * sizeof (struct timeval));
119 
120  printf("%d", ttl);
121  icmp4_packet_set_ttl(packet, ttl);
122  for (int attempt = 0; attempt < attempts_number; attempt++)
123  current_type = icmp_v4_exchange(sockfd, packet, packet_size, &address, wait_time, &printed, &times[attempt]);
124  print_times(times, attempts_number);
125  }
126 
127  return success;
128 }
