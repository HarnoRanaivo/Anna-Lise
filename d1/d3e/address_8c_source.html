27 int get_ip(const char * hostname, int family, int socktype, int protocol, struct sockaddr * address)
28 {
29  int success = -1;
30  struct addrinfo hints;
31  struct addrinfo * results;
32 
33  addrinfo_hints(&hints, family, socktype, protocol, 0);
34  success = getaddrinfo(hostname, NULL, &hints, &results);
35  if (success == 0)
36  {
37  *address = *results->ai_addr;

43  }

45  }
46 
47  return success;
48 }

70  /* Workaround pour obtenir une adresse de la machine.
71  *
72  * Sur certaines machines getaddrinfo retourne pour unique adresse 127.0.0.1
73  * lorsqu'on donne le hostname de la machine...
74  * Sur d'autres elle retourne bien les adresses des interfaces
75  * de la machine.
76  *
77  * De même avec les « vrais » ping et traceroute:
78  * `ping localhost` et `ping <hostname de la machine>` vont tous deux se faire sur
79  * 127.0.0.1, mais sur d'autres machines se feront sur deux adresses différentes.
80  * Configuration des machines ? Versions des logiciels ? ...?
81  *
82  * Avec ça, ça fonctionne partout :
83  */
84  const in_addr_t REFUSED_IPV4 = inet_addr("127.0.0.1");
85  if (address->sin_addr.s_addr == REFUSED_IPV4)
86  success = get_interface_ipv4(address);
87 
88  return success;
89 }

101 int get_interface_ipv4(struct sockaddr_in * address)
102 {
103  struct ifaddrs * results;
104  const in_addr_t REFUSED_IPV4 = inet_addr("127.0.0.1");
105 
106  int success = getifaddrs(&results);
107  if (success == 0)
108  {
109  success = -1;
110  for (const struct ifaddrs * r = results; success == -1 && r != NULL; r = r->ifa_next)
111  if (r->ifa_addr != NULL && r->ifa_addr->sa_family == AF_INET)
112  {
113  struct sockaddr_in * s = (struct sockaddr_in *) r->ifa_addr;
114  if (s->sin_addr.s_addr != REFUSED_IPV4)
115  {
116  *address = *s;
117  success = 0;
118  }
119  }
120  freeifaddrs(results);
121  }
122 
123  return success;
124 }

137 int get_interface_ipv6(struct sockaddr_in6 * address)
138 {
139  struct ifaddrs * results;
140  struct in6_addr REFUSED_IPV6;
141  inet_pton(AF_INET6, "::1", &REFUSED_IPV6);
142 
143  int success = getifaddrs(&results);
144  if (success == 0)
145  {
146  success = -1;
147  for (const struct ifaddrs * r = results; success == -1 && r != NULL; r = r->ifa_next)
148  if (r->ifa_addr != NULL && r->ifa_addr->sa_family == AF_INET6)
149  {
150  struct sockaddr_in6 * s = (struct sockaddr_in6 *) r->ifa_addr;
151  if (compare_ipv6(&s->sin6_addr, &REFUSED_IPV6) == -1)
152  {
153  char buffer[INET6_ADDRSTRLEN];
154  if (inet_ntop(AF_INET6, &s->sin6_addr, buffer, INET6_ADDRSTRLEN) != NULL)
155  if (strncmp(buffer, "fe80::", 6) != 0)
156  {
157  *address = *s;
158  success = 0;
159  }
160  }
161  }
162  freeifaddrs(results);
163  }
164 
165  return success;
166 }
