71 static inline int ping_v4(const char * hostname)
72 {
73  int sockfd;
74  struct sockaddr_in address;
75  icmp4_packet packet;
76 
77  int success = get_ipv4(hostname, IPPROTO_ICMP, &address);
78  printf("ping ");
79  print_host_v4(&address);
80  printf("\n");
81  succeed_or_die(success, 0, create_raw_socket(AF_INET, SOCK_RAW, IPPROTO_ICMP, &sockfd));
82  succeed_or_die(success, 0, icmp4_packet_init(&packet, extract_ipv4(&address)));
83  succeed_or_die(success, 0, icmp4_packet_set_length(&packet, sizeof packet));
84 
85  int sent = 0;
86  int i = 0;
87  int gotten = 0;
88  struct timeval wait_time = { 1, 0 };
89  long double min = 0.;
90  long double max = 0.;
91  long double sum = .0;
92  struct timeval ping_start = { 0, 0 };
93  struct timeval ping_end = { 0, 0 };
94  gettimeofday(&ping_start, NULL);
95  while (success == 0 && fin_des_temps == 1)
96  {
97  struct timeval start = { 0, 0 };
98  struct timeval end = { 0, 0 };
99 
100  succeed_or_die(success, 0, icmp4_packet_set_echo_seq(&packet, i));
101  gettimeofday(&start, NULL);
102  if (sendto(sockfd, &packet, sizeof packet, 0, (struct sockaddr *) &address, sizeof address) == sizeof packet)
103  {
104  sent++;
105  icmp4_packet received;
106  memset(&received, 0, sizeof received);
107  int before = gotten;
108  if (receive_icmp_v4(sockfd, &address, &wait_time, &received) == 0)
109  {
110  if (received.icmp_header.type == ICMP_ECHOREPLY
111  && received.icmp_header.un.echo.sequence == i
112  && received.icmp_header.un.echo.id == packet.icmp_header.un.echo.id
113  )
114  {
115  gotten++;
116  gettimeofday(&end, NULL);
117  struct timeval diff = diff_timeval(start, end);
118  long double rtt = extract_time(diff);
119  update_statistics(&min, &max, &sum, rtt, gotten, before);
120  print_received(&received, &address, rtt);
121 
122  if (rtt > sum / gotten * 2 || rtt < sum / gotten / 2)
123  success = -1;
124  }
125  }
126  if ((float) gotten / sent < 0.7)
127  success = -1;
128  }
129  i++;
130  sleep(1);
131  }
132  gettimeofday(&ping_end, NULL);
133  struct timeval total = diff_timeval(ping_start, ping_end);
134  print_ping_statistics(sent, gotten, min, max, sum, total, &address);
135 
136  return success;
137 
138 }
